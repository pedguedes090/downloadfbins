API Documentation

Base URL

http://localhost:3000

Facebook Video Download

POST
/api/facebook

Request Body

{
    "url": "facebook_video_url"
}

Response

{
    "success": true,
    "data": {
        "url": "original_facebook_url",
        "duration_ms": 189632,
        "sd": "sd_video_url",
        "hd": "hd_video_url",
        "title": "video_title",
        "thumbnail": "thumbnail_url",
        "filename": "local_filename.mp4",
        "localUrl": "/downloads/local_filename.mp4"
    }
}

Instagram Content Download

POST
/api/instagram

Request Body

{
    "url": "instagram_post_url"
}

Response

{
    "success": true,
    "data": {
        "results_number": 1,
        "post_info": {
            "owner_username": "username",
            "owner_fullname": "fullname",
            "is_verified": false,
            "is_private": false,
            "likes": 6,
            "is_ad": false
        },
        "media_details": [
            {
                "type": "video|image",
                "dimensions": {
                    "height": "1080",
                    "width": "1920"
                },
                "video_view_count": 1000,
                "url": "media_url",
                "thumbnail": "thumbnail_url",
                "filename": "local_filename.mp4",
                "localUrl": "/downloads/local_filename.mp4"
            }
        ]
    }
}

Error Response

{
    "success": false,
    "error": "error_message"
}

Example Usage

// Using fetch
fetch('http://localhost:3000/api/facebook', {
    method: 'POST',
    headers: {
        'Content-Type': 'application/json'
    },
    body: JSON.stringify({
        url: 'https://www.facebook.com/...'
    })
})
.then(response => response.json())
.then(data => console.log(data))
.catch(error => console.error(error));
